#include <stdio.h>

int max2(int a, int b) {
    int r;
    if (a >= b) {
        r = a;
    } else {
        r = b;
    }
    return r;
}

int main() {
    int x;
    int y;
    int m;

    x = 5;
    y = 9;
    m = max2(x, y);

    printf("max = %d\n", m);  
    return 0;
}
