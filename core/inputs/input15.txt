#include <stdio.h>

int main() {
    int a;
    int b;
    int r;

    a = 5;
    b = 10;
    r = 0;

    if (a < b) {
        r = 1 + 3;  
        a = a + 5;

        printf("a < b\n");
        printf("a = %d\n", a);
        printf("r = %d\n", r);
    } else {
        r = 0;
        printf("a >= b\n");
        printf("a = %d\n", a);
        printf("r = %d\n", r);
    }

    printf("r final = %d\n", r);

    return 0;
}
